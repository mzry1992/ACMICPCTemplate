#include <cstdio>
const int MAX1 = 300 * 16;
const int MAXN = 16;
const int MAXM = 300;
struct Link
{
	Link *l, *r, *u, *d;
	int col;
}link[MAX1 + MAXM + 1], *head;
int size;
int a[MAXN][MAXM];
Link *newLink(int col)
{
	link[size].l = link[size].r = link[size].u = link[size].d = &link[size];
	link[size].col = col;
	return &link[size ++];
}
Link *now[MAXM], *col[MAXM];
int sum[MAXM];
void init()
{
	size = 0;
}
void build(int n, int m)
{
	head = newLink(-1); 
	Link *last = head;
	for (int i = 0; i < m; ++ i)
	{
		now[i] = col[i] = newLink(i);
		last -> r = col[i];
		col[i] -> l = last;
		last = col[i];
		sum[i] = 0;
	}
	head -> l = last;
	last -> r = head;
	for (int i = 0; i < n; ++ i)
	{
		Link *first = 0, *last = 0;
		for (int j = 0; j < m; ++ j)
			if (a[i][j])
			{
				Link *p = newLink(j);
				++ sum[j];
				now[j] -> d = p;
				p -> u = now[j];
				now[j] = p;
				if (!last)
					last = first = p;
				else
				{
					last -> r = p;
					p -> l = last;
					last = p;
				}
			}
		if (!first)
			continue;
		first -> l = last;
		last -> r = first;
	}
	for (int i = 0; i < m; ++ i)
	{
		col[i] -> u = now[i];
		now[i] -> d = col[i];
	}
}
void remove(Link *x)
{
	x -> l -> r = x -> r;
	x -> r -> l = x -> l;
	for (Link *p = x -> d; p != x; p = p -> d)
		for (Link *q = p -> r; q != p; q = q -> r)
		{
			q -> u -> d = q -> d;
			q -> d -> u = q -> u;
			-- sum[q -> col];
		}
}
void resume(Link *x)
{
	for (Link *p = x -> u; p != x; p = p -> u)
		for (Link *q = p -> l; q != p; q = q -> l)
		{
			q -> d -> u = q;
			q -> u -> d = q;
			++ sum[q -> col];
		}
	x -> r -> l = x;
	x -> l -> r = x;
}
bool dfs(int step)
{
	printf("%d\n", step);
	if (head -> r == head)
		return 1;
	if (!step)
		return 0;
	Link *idx = head -> r;
	for (Link *p = idx -> r; p != head; p = p -> r)
		if (sum[p -> col] < sum[idx -> col])
			idx = p;
	remove(idx);
	for (Link *p = idx -> d; p != idx; p = p -> d)
	{
		for (Link *q = p -> r; q != p; q = q -> r)
			remove(col[q -> col]);
		if (dfs(step - 1))
			return 1;
		for (Link *q = p -> l; q != p; q = q -> l)
			resume(col[q -> col]);
	}
	resume(idx);
	return 0;
}
int main()
{
	int n, m;
	while (scanf("%d%d", &n, &m) != EOF)
	{
		for (int i = 0; i < n; ++ i)
			for (int j = 0; j < m; ++ j)
				//scanf("%d", &a[i][j]);
				a[i][j] = 0;
		init();
		build(n, m);
		if (dfs(n))
			puts("Yes, I found it");
		else
			puts("It is impossible");
	}
	return 0;
}

void remove1(int col)
{
	int i,j;
	L[R[col]]=L[col];
	R[L[col]]=R[col];
	for(i=D[col];i!=col;i=D[i])
	{
		L[R[i]]=L[i];
		R[L[i]]=R[i];
	}
}
void remove2(int col)
{
	int i,j;
	L[R[col]]=L[col];
	R[L[col]]=R[col];
	for(i=D[col];i!=col;i=D[i])
	{
		for(j=R[i];j!=i;j=R[j])
		{
			U[D[j]]=U[j];
			D[U[j]]=D[j];
			--nk[C[j]];
		}
	}
}
void resume1(int col)
{
	int i,j;
	for(i=U[col];i!=col;i=U[i])
	{
		L[R[i]]=i;
		R[L[i]]=i;
	}
	L[R[col]]=col;
	R[L[col]]=col;
}
void resume2(int col)
{
	int i,j;
	for(i=U[col];i!=col;i=U[i])
	{
		for(j=L[i];j!=i;j=L[j])
		{
			++nk[C[j]];
			U[D[j]]=j;
			D[U[j]]=j;
		}
	}
	L[R[col]]=col;
	R[L[col]]=col;
}

bool DLX(int d,int deep)
{
	if(d+h()>deep) return false;
	if(R[head]==head||R[head]>2*n)
		return true;
	if(d>=deep)
		return false;
	int col,ma=inf;
	int i,j;
	for(i=R[head];i!=head&&i<=2*n;i=R[i])
		if(nk[i]<ma)
		{
			col=i;
			ma=nk[i];
		}
	remove1(col);
	for(i=D[col];i!=col;i=D[i])
	{
		int flag=1;
		for(j=R[i];;j=R[j])
		{
			if(j==R[i]&&!flag)
				break;
			U[D[j]]=U[j];
			D[U[j]]=D[j];
			if(C[j]>2*n)
				remove2(C[j]);
			else
				remove1(C[j]);
			flag=0;
		}
		if(DLX(d+1,deep))
			return true;
		flag=1;
		for(j=L[i];;j=L[j])
		{
			if(j==L[i]&&!flag)
				break;
			if(C[j]>2*n)
				resume2(C[j]);
			else
				resume1(C[j]);
			U[D[j]]=j;
			D[U[j]]=j;
			flag=0;
		}
	}
	resume1(col);
	return false;
}
