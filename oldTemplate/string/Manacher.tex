\subsection{Manacher}
	\begin{lstlisting}[language=c++]
const int maxn = 110000;

char Ma[maxn*2];
int Mp[maxn*2];
void Manacher(char s[],int len)
{
	int l = 0;
	Ma[l++] = '.';
	Ma[l++] = ',';
	for (int i = 0;i < len;i++)
	{
		Ma[l++] = s[i];
		Ma[l++] = ',';
	}
	Ma[l] = 0;
	int pnow = 0,pid = 0;
	for (int i = 1;i < l;i++)
	{
		if (pnow > i)
			Mp[i] = min(Mp[2*pid-i],pnow-i);
		else
			Mp[i] = 1;
		for (;Ma[i-Mp[i]] == Ma[i+Mp[i]];Mp[i]++);
		if (i+Mp[i] > pnow)
		{
			pnow = i+Mp[i];
			pid = i;
		}
	}
}
