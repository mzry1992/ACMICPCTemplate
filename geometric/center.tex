Point CenterOfPolygon(Point poly[],int n)
{
	Point p, p0, p1, p2, p3;
	double m, m0;
	p1 = poly[0];
	p2 = poly[1];
	p.x = p.y = m = 0;
	for (int i = 2; i < n; i++)
	{
		p3 = poly[i];
		p0.x = (p1.x + p2.x + p3.x) / 3.0;
		p0.y = (p1.y + p2.y + p3.y) / 3.0;
		m0 = p1.x*p2.y+p2.x*p3.y+p3.x*p1.y-p1.y*p2.x-p2.y*p3.x-p3.y*p1.x;
		if (cmp(m + m0,0.0) == 0)
		m0 += eps;
		p.x = (m * p.x + m0 * p0.x) / (m + m0);
		p.y = (m * p.y + m0 * p0.y) / (m + m0);
		m = m + m0;
		p2 = p3;
	}
	return p;
}
