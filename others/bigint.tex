\subsection{高精度数}
	支持乘以整数和加法。\\
	\begin{lstlisting}[language=c++]
struct BigInt
{
	const static int mod = 100000000;
	int a[600],len;
	BigInt (){}
	BigInt (int v)
	{
		len = 0;
		do
		{
			a[len++] = v%mod;
			v /= mod;
		}while(v);
	}
	BigInt operator *(const int& b) const
	{
		BigInt res;
		res.len = len;
		for (int i = 0; i <= len; ++i)
			res.a[i] = 0;
		for (int i = 0; i < len; ++i)
		{
			res.a[i] += a[i]*b;
			res.a[i+1] += res.a[i]/mod;
			res.a[i] %= mod;
		}
		if (res.a[len] > 0)  res.len++;
		return res;
	}
	BigInt operator +(const BigInt& b) const
	{
		BigInt res;
		res.len = max(len,b.len);
		for (int i = 0; i <= res.len; ++i)
			res.a[i] = 0;
		for (int i = 0; i < res.len; ++i)
		{
			res.a[i] += ((i < len)?a[i]:0)+((i < b.len)?b.a[i]:0);
			res.a[i+1] += res.a[i]/mod;
			res.a[i] %= mod;
		}
		if (res.a[res.len] > 0)  res.len++;
		return res;
	}
	void output()
	{
		printf("%d",a[len-1]);
		for (int i = len-2; i >= 0; --i)
			printf("%08d",a[i]);
		printf("\n");
	}
};
	\end{lstlisting}
